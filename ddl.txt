CREATE SCHEMA `lms-database`;
use `lms-database`;

CREATE TABLE `lms-database`.`user_details` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `email` VARCHAR(45) NOT NULL,
  `first_name` VARCHAR(45) NOT NULL,
  `last_name` VARCHAR(45) NOT NULL,
  `password` VARCHAR(45) NOT NULL,
  `contact_number` VARCHAR(45) NOT NULL,
  `verified` VARCHAR(45) NOT NULL,
  `creater_stamp` DATETIME NOT NULL,
  `creater_user` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `email_UNIQUE` (`email` ASC) VISIBLE);


CREATE TABLE `hired_candidate` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `first_name` VARCHAR(45) NOT NULL,
  `middle_name` VARCHAR(45) NULL,
  `last_name` VARCHAR(45) NOT NULL,
  `email_id` VARCHAR(45) NOT NULL,
  `degree` VARCHAR(45) NOT NULL,
  `hired_city` VARCHAR(45) NOT NULL,
  `hired_date` DATE NOT NULL,
  `mobile_number` VARCHAR(45) NOT NULL,
  `permanent_pincode` VARCHAR(45) NOT NULL,
  `hired_lab` VARCHAR(45) NOT NULL,
  `attitude` VARCHAR(45) NOT NULL,
  `communication_remark` VARCHAR(45) NOT NULL,
  `knowledge_remark` VARCHAR(45) NOT NULL,
  `aggregate_remark` VARCHAR(45) NOT NULL,
  `status` VARCHAR(45) NOT NULL DEFAULT 'Didn\'t Respond',
  `creator_stamp` DATETIME NOT NULL,
  `creator_user` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `email_id_UNIQUE` (`email_id` ASC) VISIBLE);


CREATE TABLE `fellowship_candidate` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `first_name` VARCHAR(45) NOT NULL,
  `middle_name` VARCHAR(45) NULL,
  `last_name` VARCHAR(45) NOT NULL,
  `email_id` VARCHAR(45) NOT NULL,
  `degree` VARCHAR(45) NOT NULL,
  `hired_city` VARCHAR(45) NOT NULL,
  `hired_date` DATE NOT NULL,
  `mobile_number` VARCHAR(45) NOT NULL,
  `permanent_pincode` INT NOT NULL,
  `hired_lab` VARCHAR(45) NOT NULL,
  `attitude` VARCHAR(45) NOT NULL,
  `communication_remark` VARCHAR(45) NOT NULL,
  `knowledge_remark` VARCHAR(45) NOT NULL,
  `aggregate_remark` VARCHAR(45) NOT NULL,
  `creator_stamp` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `creator_user` VARCHAR(45) NOT NULL,
  `birth_date` DATE NOT NULL,
  `is_birth_date_verified` VARCHAR(45) NOT NULL DEFAULT 'false',
  `parent_name` VARCHAR(45) NOT NULL,
  `parent_occupation` VARCHAR(45) NOT NULL,
  `parents_mobile_number` VARCHAR(45) NOT NULL,
  `parents_annual_salary` INT NOT NULL,
  `local_address` VARCHAR(150) NOT NULL,
  `permanent_address` VARCHAR(150) NOT NULL,
  `photo_path` VARCHAR(100) NOT NULL,
  `joining_date` DATE NOT NULL,
  `candidate_status` VARCHAR(45) NOT NULL DEFAULT 'Pending',
  `personal_information` VARCHAR(150) NULL,
  `bank_information` VARCHAR(45) NULL,
  `educational_information` VARCHAR(45) NULL,
  `document_status` VARCHAR(45) NULL,
  `remark` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `email_id_UNIQUE` (`email_id` ASC) VISIBLE);

CREATE TABLE `candidate_bank_details` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `candidate_id` INT NOT NULL,
  `name` VARCHAR(45) NOT NULL,
  `account_number` VARCHAR(45) NOT NULL,
  `is_account_num_verified` VARCHAR(45) NOT NULL DEFAULT 'false',
  `ifsc_code` VARCHAR(45) NOT NULL,
  `is_ifsc_code_verified` VARCHAR(45) NOT NULL DEFAULT 'false',
  `pan_number` VARCHAR(45) NOT NULL,
  `is_pan_number_verified` VARCHAR(45) NOT NULL DEFAULT 'false',
  `aadhaar_num` VARCHAR(45) NOT NULL,
  `is_aadhaar_num_verified` VARCHAR(45) NOT NULL DEFAULT 'false',
  `creator_stamp` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `creator_user` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `fk_candidate_id_idx` (`candidate_id` ASC) VISIBLE,
  UNIQUE INDEX `candidate_id_UNIQUE` (`candidate_id` ASC) VISIBLE,
  CONSTRAINT `fk_candidate_id`
    FOREIGN KEY (`candidate_id`)
    REFERENCES `lms-database`.`fellowship_candidate` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

CREATE TABLE `lms-database`.`candidate_documents` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `candidate_id` INT NOT NULL,
  `document_type` VARCHAR(45) NOT NULL,
  `document_path` VARCHAR(45) NOT NULL,
  `status` VARCHAR(45) NOT NULL DEFAULT 'Pending',
  `creator_stamp` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `creator_user` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `fk_candidate_id_documents_idx` (`candidate_id` ASC) VISIBLE,
  UNIQUE INDEX `candidate_id_UNIQUE` (`candidate_id` ASC) VISIBLE,
  CONSTRAINT `fk_candidate_id_documents`
    FOREIGN KEY (`candidate_id`)
    REFERENCES `lms-database`.`fellowship_candidate` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);


CREATE TABLE `company` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(45) NOT NULL,
  `address` VARCHAR(100) NOT NULL,
  `location` VARCHAR(100) NOT NULL,
  `status` VARCHAR(45) NOT NULL,
  `creator_stamp` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `creator_user` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`));

CREATE TABLE `lms-database`.`tech_stack` (
  `id` INT NOT NULL,
  `tech_name` VARCHAR(45) NOT NULL,
  `image_path` VARCHAR(45) NOT NULL,
  `framework` VARCHAR(45) NOT NULL,
  `current_status` VARCHAR(45) NOT NULL,
  `creator_stamp` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `creator_user` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`));
